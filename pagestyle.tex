	\rhead{\includegraphics[width=0.30\textwidth]{logos/logo-imt.jpg}}
	\lfoot{\includegraphics[width=0.28\textwidth]{logos/logo-univ-ups.png}}
	\rfoot{\includegraphics[width=0.28\textwidth]{logos/logo-irit.jpg}}
	\fancyhead[C]{} % On enlève les informations du header
	\fancyfoot[C]{} % On enlève les informations du footer
	\renewcommand{\headrulewidth}{0pt} % On enlève la ligne du header
}
\fancypagestyle{body}{%
	\restoregeometry
	\pagestyle{fancy}
	\fancyhf{}
	
	\if@twoside
		\fancyhead[LO]{\thesection. \rightmark} %rightmark = section
		\fancyhead[RE]{\textsc{Chapter} \thechapter. \textsc{\leftmark}} %leftmark=chapter
		\fancyhead[RO]{\thepage}
		\fancyhead[LE]{\thepage}
	\else
		\fancyhead[L]{\textsc{Chapter} \thechapter. \textsc{\leftmark}}
		\fancyhead[R]{\thesection. \rightmark}
		\fancyfoot[C]{\thepage}
	\fi

	\renewcommand*{\chaptermark}[1]{\markboth{##1}{}}
	\renewcommand*{\sectionmark}[1]{\markright{##1}{}}
	\renewcommand\headrulewidth{.1pt}
	\setlength{\parskip}{0.2cm} % Espace entre paragraphes
%	\renewcommand{\arraystretch}{1.5} % Espace entre cases d'un tableau
}
\fancypagestyle{plain}{
	\if@twoside
		\pagestyle{body}
		\fancyhead[LO]{\rightmark}
	\else
		\pagestyle{body}
		\fancyhf{}
		\renewcommand\headrulewidth{0pt}
	\fi
}
\fancypagestyle{appendix}{%
	\restoregeometry
	% On nettoie les headers et footers existants de "fancy"
	\fancyhf{}
	% Pour le header
	\fancyhead[RE]{Appendix}
	\fancyhead[LO]{\thechapter. \leftmark} %leftmark = chapter
	\fancyhead[RO]{\thepage} %
	\fancyhead[LE]{\thepage} %
	\renewcommand*{\chaptermark}[1]{\markboth{##1}{}}
	\renewcommand\headrulewidth{.1pt}
	% Pour le footer
	%\fancyfoot[C]{\thepage}
	\renewcommand\footrulewidth{0pt}
}
